00581 template <typename ParameterModel>
00582 template< typename MzExtractor >
00583 void PeakParameterFwhm<ParameterModel>::learn_(const std::vector<std::pair<typename MzExtractor::result_type, typename MzExtractor::result_type> >& pairs) {
00584     using namespace vigra;
00585     typedef std::vector<std::pair<typename MzExtractor::result_type, typename MzExtractor::result_type> > MzWidthPairs_;
00586     psf_precondition(pairs.empty() == false, "PeakParameterFwhm::learn_(): Called with empty input vector. This is not supposed to happen. A bug in the code preceding the call of learn_ probably caused it.");
00587 
00588     // We now fit the PeakParameterModel to the spectrum using linear regression.    
00589     // We minimize the residue |A*x - b|^2.
00590     // b is a column vector with all measured widths.
00591     // x is a column vector with the model parameters, for example (a, b)^T.
00592     // A is a matrix such that A*x resembles the original model in every row for the
00593     // corresponding mz value.
00594     // x gets optimized.
00595 
00596     /* Construct A and b */
00597 
00598     // Models with no parameter should not implement the numberOfParameters() function.
00599     // Calling it here while trying to learn this non-existing parameters would cause a
00600     // desirable compile time error. Nevertheless, we have to guard against wrongly implemented
00601     // ParameterModels.
00602     psf_invariant(this->ParameterModel::numberOfParameters() > 0, "PeakParameterFwhm::learn_(): Number of model parameters is not greater than zero.");
00603     
00604     // A: #rows is number of measured pairs; #columns is dimension of parameter space
00605     linalg::Matrix<double> A(pairs.size(), this->ParameterModel::numberOfParameters());
00606     // b: column vector with as many elements as measured pairs
00607     linalg::Matrix<double> b(pairs.size(), 1);
00608     GeneralizedSlope slope;
00609 
00610     // Calc GeneralizedSlope for every measured pair and store it as rows of A
00611     // Store the measured width in b
00612     for(typename MzWidthPairs_::size_type pairIndex = 0; pairIndex < pairs.size(); ++pairIndex) {
00613         slope = this->ParameterModel::slopeInParameterSpaceFor(pairs[pairIndex].first);
00614 
00615         // copy the slope into a row of A
00616         // (we ignore the bias, because it can't be optimized.)
00617         psf_invariant((slope.size()-1) == static_cast<GeneralizedSlope::size_type>(A.columnCount()), "PeakParameterFwhm::learn_(): Generalized slope has different dimension than the space, it is living in.");        
00618         for(linalg::Matrix<double>::difference_type_1 column = 0; column < A.columnCount(); ++column) {      
00619             A(static_cast<linalg::Matrix<double>::difference_type_1>(pairIndex), column) = slope.at(static_cast<GeneralizedSlope::size_type>(column));   
00620         }       
00621 
00622         b(static_cast<linalg::Matrix<double>::difference_type_1>(pairIndex), 0) = pairs[pairIndex].second;    
00623     }
00624 
00625     /* do least squares */
00626     // result: the optimized parameters
00627     // Note, that we don't include the bias.
00628     linalg::Matrix<double> x(this->ParameterModel::numberOfParameters(),1);
00629     
00630     // We have to enforce a positive FWHM for positive mz values, so we use a non-negative
00631     // least squares with x guaranteed to be non-negative. The model then has to yield
00632     // positive values too, of course. But that is in the responsibility of the caller. 
00633     linalg::nonnegativeLeastSquares(A, b, x);
00634 
00635     /* set the fitted parameters */
00636     for(linalg::Matrix<double>::difference_type_1 index = 0; index < x.rowCount(); ++index) {
00637         PSF_LOG(logDEBUG2) << "PeakParameterFwhm::learn_(): Parameter " << index << " found: " <<  x(index, 0);
00638         
00639         psf_invariant(index >= 0, "PeakParameterFwhm::learn_(): index may not be negative before calling setParameter(index,0).");
00640         this->ParameterModel::setParameter(index, x(static_cast<unsigned>(index), 0));    
00641     }
00642 }
