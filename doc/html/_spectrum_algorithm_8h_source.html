00235 template<typename FwdIter, typename Compare>
00236 std::pair<FwdIter, FwdIter> findBump(FwdIter first, FwdIter last, Compare comp) {
00237     // First, let's assume the bump starts right on the first element.
00238     FwdIter leftEdge = first;
00239     // Furthermore, we are starting out on the first element, too.
00240     FwdIter currentElement = first;
00241 
00242     // Since we are working with ForwardIterators, we cannot rewind and have to store the
00243     // next position separately
00244     const FwdIter onePastLastElement = last;
00245     FwdIter onePastCurrentElement = currentElement; std::advance(onePastCurrentElement, 1);   
00246 
00247     // state of our search
00248     bool onIncreasingSlope = false;
00249     bool foundBumpTop = false;
00250 
00251     // Now let's go through the sequence element by element.
00252     while(onePastCurrentElement != onePastLastElement) {
00253         psf_invariant(std::distance(currentElement, onePastCurrentElement) == 1, "findBump(): Distance between current and next elment is not 1.");
00254         
00255         /* Current element is smaller than its right neighbor. */      
00256         if(comp(*currentElement, *(onePastCurrentElement))) {
00257             // We already passed the top of a bump and are finished!
00258             if(foundBumpTop) {
00259                 // currentElement points to the right edge of the bump
00260                 break;
00261             }
00262             else {
00263                 // We are on the bottom of an increasing slope.
00264                 if(!onIncreasingSlope) {
00265                     onIncreasingSlope = true;
00266                     // That's were bumps are starting!
00267                     leftEdge = currentElement;
00268                 }
00269             }
00270         }
00271         
00272         /* Current element is bigger than its right neighbor. */
00273         else if(comp(*(onePastCurrentElement), *currentElement)) { 
00274             // Were we on an increasing slope?
00275             if(onIncreasingSlope) {
00276                 // So we are on the top of the bump!
00277                 foundBumpTop = true;
00278             }
00279             // We are on a decreasing slope.
00280             else {
00281                 onIncreasingSlope = false;
00282             }
00283         }
00284 
00285         /* Current element and its neighbor are equal. */
00286         else {
00287             // Great, we are finished!
00288             if(foundBumpTop) {
00289                 // currentElement points to the right edge of our bump.
00290                 break;
00291             }
00292             
00293             // That's bad. We have to start our search again.
00294             else {            
00295                 leftEdge = onePastCurrentElement;
00296                 onIncreasingSlope = false;
00297             }
00298         }
00299 
00300         ++currentElement;
00301         ++onePastCurrentElement;
00302     }
00303     
00304     if(foundBumpTop) { 
00305         // we return [leftEdge, rightEdge]
00306         return std::make_pair(leftEdge, currentElement);
00307     }
00308     else {
00309         return std::make_pair(last, last);
00310     }
00311 }

00325     // The result
00326     std::vector<std::pair<Mz, Mz> > widths;
00327 
00328     // Check for empty spectrum or only one element
00329     if((last - first) < 1) {
00330         return widths;
00331     }
00332 
00333     // prerequisites    
00334     const double requiredLowness = 1. - fraction;
00335     std::pair<FwdIter, FwdIter> bump;

00339     LessByExtractor< typename IntensityExtractor::element_type, IntensityExtractor > comp(get_int);
00340     
00341     // go through all bumps in the spectrum */       
00342     while(first < last) {
00343         bump = findBump(first, last, comp);
00344         // No new bump found
00345         if (bump.first == last) {
00346             break;
00347         }
00348         
00349         // calc full width if bump is low enough and has a minimal height
00350         psf_invariant(bump.first <= bump.second && bump.second < last, "Bump in illegal state.");
00351         bumpHeight = get_int(*std::max_element(bump.first, bump.second + 1, comp));       
00352         if(SpectralPeak::lowness(get_int, bump.first, bump.second) >= requiredLowness && bumpHeight >= minimalPeakHeight) {
00353             // we don't have to try for exceptions here, because a bump fulfills the preconditions
00354     width = SpectralPeak::fullWidthAtFractionOfMaximum(get_mz, get_int, bump.first, bump.second, fraction);
00355             positionOfMaximum = get_mz(*std::max_element(bump.first, bump.second + 1, comp));
00356             PSF_LOG(logDEBUG) << "measureFullWidths(): Measured peak (mz | width): (" << positionOfMaximum << " | " << width << ")";
00357             widths.push_back(std::make_pair(positionOfMaximum, width));
00358         }
00359         
00360         // last element of the bump may be the first of the next one

00399     FwdIter rightMinimum = min_element(maximum, last, comp);
